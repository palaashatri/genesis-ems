```sh

  cd source_code/
# 2. For Running the main.py file in source_code directory
  python3 main.py
# 3. For Running the test_all.py file source_code directory
  python3 -m pytest
# 4. For Running the pytest verbose command in source_code directory
  py.test -v  
```
